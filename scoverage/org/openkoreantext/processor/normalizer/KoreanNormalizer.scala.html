36 object KoreanNormalizer {
37   private[this] val EXTENTED_KOREAN_REGEX = """([ㄱ-ㅣ가-힣]+)""".r
38   private[this] val KOREAN_TO_NORMALIZE_REGEX = """([가-힣]+)(ㅋ+|ㅎ+|[ㅠㅜ]+)""".r
39   protected[processor] val REPEATING_CHAR_REGEX: Regex = """(.)\1{3,}|[ㅠㅜ]{3,}""".r
40   private[this] val REPEATING_2CHAR_REGEX = """(..)\1{2,}""".r
41 
42   private[this] val WHITESPACE_REGEX = """\s+""".r
43 
44   private[this] case class Segment(text: String, matchData: Option[Match])

46   private[this] val CODA_N_EXCPETION = "은는운인텐근른픈닌든던".toSet

48   /**
49    * Normalize Korean CharSequence text
50    * ex) 하댘ㅋㅋㅋ -> 하대, 머구뮤ㅠㅠㅠ -> 머굼
51    * 하즤 -> 하지
52    *
53    * @param input input CharSequence
54    * @return normalized CharSequence
55    */
56   def normalize(input: CharSequence): CharSequence = {
57     EXTENTED_KOREAN_REGEX.replaceAllIn(input, m => normalizeKoreanChunk(m.group(0)).toString)
58   }
59 
60   private[this] def normalizeKoreanChunk(input: CharSequence): CharSequence = {
61     // Normalize endings: 안됔ㅋㅋㅋ -> 안돼ㅋㅋ
62     val endingNormalized = KOREAN_TO_NORMALIZE_REGEX.replaceAllIn(
63       input, m => processNormalizationCandidate(m).toString
64     )
65 
66     // Normalize repeating chars: ㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋㅋ -> ㅋㅋㅋ
67     val exclamationNormalized = REPEATING_CHAR_REGEX.replaceAllIn(
68       endingNormalized, m => {
69         Matcher.quoteReplacement(m.group(0).take(3).toString)
70       }
71     )
72     // Normalize repeating chars: 훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍훌쩍 -> 훌쩍훌쩍
73     val repeatingNormalized = REPEATING_2CHAR_REGEX.replaceAllIn(
74       exclamationNormalized, m => {
75         Matcher.quoteReplacement(m.group(0).take(4).toString)
76       }
77     )
78 
79     // Coda normalization (명사 + ㄴ 첨가 정규화): 소린가 -> 소리인가
80     val codaNNormalized = normalizeCodaN(repeatingNormalized)
81     // Typo correction: 하겟다 -> 하겠다
82     val typoCorrected = correctTypo(codaNNormalized)
83     // Spaces, tabs, new lines are replaced with a single space.
84     WHITESPACE_REGEX.replaceAllIn(typoCorrected, " ")
85   }

87   protected[processor] def correctTypo(chunk: CharSequence): CharSequence =
88     typoDictionaryByLength.foldLeft(chunk) {
89       case (output: String, (wordLen: Int, typoMap: Map[String, String])) =>
90         output.sliding(wordLen).foldLeft(output) {
91           case (sliceOutput: String, slice: String) if typoMap.contains(slice) =>
92             sliceOutput.replaceAll(slice, typoMap(slice))
93           case (sliceOutput: String, slice: String) =>
94             sliceOutput
95         }
96     }
97 
98   protected[processor] def normalizeCodaN(chunk: CharSequence): CharSequence = {
99     if (chunk.length < 2) return chunk
100 
101     val lastTwo = chunk.subSequence(chunk.length() - 2, chunk.length())
102     val last = chunk.charAt(chunk.length() - 1)
103 
104     val lastTwoHead = lastTwo.charAt(0)
105 
106     // Exception cases
107     if (koreanDictionary.get(Noun).contains(chunk) ||
108         koreanDictionary.get(Conjunction).contains(chunk) ||
109         koreanDictionary.get(Adverb).contains(chunk) ||
110         koreanDictionary.get(Noun).contains(lastTwo) ||
111         lastTwoHead < '가' || lastTwoHead > '힣' ||
112         CODA_N_EXCPETION.contains(lastTwoHead)
113     ) {
114       return chunk
115     }
116 
117     if (KoreanPos.Predicates.contains(KoreanTokenizer.tokenize(chunk).head.pos)) {
118       return chunk
119     }
120 
121     val hc = decomposeHangul(lastTwoHead)
122 
123     val newHead = new StringBuilder()
124         .append(chunk.subSequence(0, chunk.length() - 2))
125         .append(composeHangul(hc.onset, hc.vowel))
126 
127     if (hc.coda == 'ㄴ' &&
128         (last == '데' || last == '가' || last == '지') &&
129         koreanDictionary.get(Noun).contains(newHead)
130     ) {
131       newHead + "인" + last
132     } else {
133       chunk
134     }
135   }
136 
137   private[this] def processNormalizationCandidate(m: Match): CharSequence = {
138     val chunk = m.group(1)
139     val toNormalize = m.group(2)
140 
141     val normalizedChunk = if (koreanDictionary.get(Noun).contains(chunk) ||
142         koreanDictionary.get(Eomi).contains(chunk.takeRight(1)) ||
143         koreanDictionary.get(Eomi).contains(chunk.takeRight(2))) {
144       chunk
145     } else {
146       normalizeEmotionAttachedChunk(chunk, toNormalize)
147     }
148     normalizedChunk + toNormalize
149   }
150 
151   private[this] def normalizeEmotionAttachedChunk(s: CharSequence, toNormalize: CharSequence): CharSequence = {
152     val init = s.subSequence(0, s.length() - 1)
153     val secondToLastDecomposed = init match {
154       case si: CharSequence if si.length > 0 =>
155         val hc = decomposeHangul(si.charAt(si.length() - 1))
156         if (hc.coda == ' ') Some(hc) else None
157       case _ => None
158     }
159 
160     decomposeHangul(s.charAt(s.length() - 1)) match {
161       case hc: HangulChar if hc.coda == 'ㅋ' || hc.coda == 'ㅎ' =>
162         new StringBuilder()
163             .append(init)
164             .append(composeHangul(hc.onset, hc.vowel))
165       case HangulChar(o: Char, v: Char, ' ') if secondToLastDecomposed.isDefined &&
166           (v == toNormalize.charAt(0)) &&
167           Hangul.CODA_MAP.contains(o) =>
168         val hc = secondToLastDecomposed.get
169         new StringBuilder()
170             .append(init.subSequence(0, init.length() - 1))
171             .append(composeHangul(hc.onset, hc.vowel, o))
172       case _ => s
173     }
174   }
