21 /**
22  * Hangul analysis helper. One Hangul character can be decomposed to consonants and a vowel.
23  * This object helps analyze Korean character by consonant and vowel level.

74   /**
75    * Decompose a Korean character to onset(초성), vowel(중성), and coda(종성).
76    *
77    * @param c A Korean character
78    * @return (onset: Char, vowel: Char, coda: Char)
79    */
80   def decomposeHangul(c: Char): HangulChar = {
81     require(!ONSET_MAP.contains(c) && !VOWEL_MAP.contains(c) && !CODA_MAP.contains(c),
82       "Input character is not a valid Korean character")
83 
84     val u = c - HANGUL_BASE
85     HangulChar(ONSET_LIST(u / ONSET_BASE),
86       VOWEL_LIST((u % ONSET_BASE) / VOWEL_BASE),
87       CODA_LIST(u % VOWEL_BASE))
88   }
89 
90   /**
91    * Check if a Korean character has a coda.
92    *
93    * @param c A Korean character
94    * @return true if the character has a coda.
95    */
96   def hasCoda(c: Char): Boolean = (c - HANGUL_BASE) % VOWEL_BASE > 0
97 
98   /**
99    * Compose a Korean character from the provided onset(초성), vowel(중성), and coda(종성).
100    *
101    * @param onset 초성
102    * @param vowel 중성
103    * @param coda 종성
104    * @return A Korean character
105    */
106   def composeHangul(onset: Char, vowel: Char, coda: Char = ' '): Char = {
107     require(onset != ' ' && vowel != ' ', "Input characters are not valid")
108 
109     (HANGUL_BASE +
110         (ONSET_MAP(onset) * ONSET_BASE) +
111         (VOWEL_MAP(vowel) * VOWEL_BASE) +
112         CODA_MAP(coda)).toChar
113   }
114 
115   /**
116    * Compose a Korean character from the provided onset(초성), vowel(중성), and coda(종성).
117    *
118    * @param hc HangulChar(onset, vowel, coda)
119    * @return A Korean character
120    */
121   def composeHangul(hc: HangulChar): Char =
122     composeHangul(hc.onset, hc.vowel, hc.coda)
