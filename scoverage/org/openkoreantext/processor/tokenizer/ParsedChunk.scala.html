27 object ParsedChunk {
28   val suffixes = Set(Suffix, Eomi, Josa, PreEomi)
29   val preferredBeforeHaVerb = Set(Noun, ProperNoun, VerbPrefix)
30 }
31 
32 /**
33   * A candidate parse for a chunk.
34   *
35   * @param posNodes Sequence of KoreanTokens.
36   * @param words Number of words in this candidate parse.
37   */
38 case class ParsedChunk(posNodes: Seq[KoreanToken], words: Int,
39     profile: TokenizerProfile = TokenizerProfile.defaultProfile) {
40 
41   // Using lazy val to cache the score
42   lazy val score: Float = countTokens * profile.tokenCount +
43       countUnknowns * profile.unknown +
44       words * profile.wordCount +
45       getUnknownCoverage * profile.unknownCoverage +
46       getFreqScore * profile.freq +
47       countPos(Unknown) * profile.unknownPosCount +
48       isExactMatch * profile.exactMatch +
49       isAllNouns * profile.allNoun +
50       isPreferredPattern * profile.preferredPattern +
51       countPos(Determiner) * profile.determinerPosCount +
52       countPos(Exclamation) * profile.exclamationPosCount +
53       isInitialPostPosition * profile.initialPostPosition +
54       isNounHa * profile.haVerb +
55       hasSpaceOutOfGuide * profile.spaceGuidePenalty +
56       josaMismatched * profile.josaUnmatchedPenalty
57 
58   def countUnknowns: Int = this.posNodes.count { p: KoreanToken => p.unknown }
59 
60   def countTokens: Int = this.posNodes.size
61 
62   def isInitialPostPosition: Int = if (suffixes.contains(this.posNodes.head.pos)) 1 else 0
63 
64   def isExactMatch: Int = if (this.posNodes.size == 1) 0 else 1
65 
66   def hasSpaceOutOfGuide: Int = if (profile.spaceGuide.isEmpty) {
67     0
68   } else {
69     this.posNodes
70         .filter{p: KoreanToken => !suffixes.contains(p.pos)}
71         .count {
72           p: KoreanToken => !profile.spaceGuide.contains(p.offset)
73         }
74   }
75 
76 
77   def isAllNouns: Int = if (this.posNodes.exists(
78     t => t.pos != Noun && t.pos != ProperNoun)) 1
79   else 0
80 
81   def isPreferredPattern: Int = if (
82     posNodes.size == 2 && profile.preferredPatterns.contains(posNodes.map(_.pos))
83   ) 0
84   else 1
85 
86   def isNounHa: Int = if (this.posNodes.size >= 2
87       && preferredBeforeHaVerb.contains(this.posNodes.head.pos)
88       && this.posNodes(1).pos == Verb
89       && (this.posNodes(1).text.startsWith("하") || this.posNodes(1).text.startsWith("해")))
90     0
91   else
92     1
93 
94   def posTieBreaker: Int = this.posNodes.map(_.pos.id).sum
95 
96   def getUnknownCoverage: Int = this.posNodes.foldLeft(0) {
97     case (sum, p: KoreanToken) => if (p.unknown) sum + p.text.length else sum
98   }
99 
100   def getFreqScore: Float = this.posNodes.foldLeft(0f) {
101     case (output: Float, p: KoreanToken) if p.pos == Noun || p.pos == ProperNoun =>
102       output + (1f - koreanEntityFreq.getOrDefault(p.text, 0f))
103     case (output: Float, p: KoreanToken) => output + 1.0f
104   } / this.posNodes.size
105 
106   def ++(that: ParsedChunk): ParsedChunk = {
107     ParsedChunk(this.posNodes ++ that.posNodes, this.words + that.words, profile)
108   }
109 
110   def countPos(pos: KoreanPos): Int = this.posNodes.count { p: KoreanToken => p.pos == pos }
111 
112   def josaMismatched: Int = {
113     val mismatched = this.posNodes.sliding(2).exists{
114       case tokenPair: Seq[KoreanToken] if tokenPair.head.pos == KoreanPos.Noun
115         && tokenPair.last.pos == KoreanPos.Josa =>
116         if (Hangul.hasCoda(tokenPair.head.text.last)) {
117           val nounEnding = Hangul.decomposeHangul(tokenPair.head.text.last)
118           (nounEnding.coda != 'ㄹ' && tokenPair.last.text.head == '로') ||
119             Set("는", "를", "다").contains(tokenPair.last.text)
120         } else {
121           tokenPair.last.text.head == '으' || Set("은", "을", "이").contains(tokenPair.last.text)
122         }
123       case _ =>
124         false
125     }
126     if (mismatched) 1 else 0
127   }
128 }
