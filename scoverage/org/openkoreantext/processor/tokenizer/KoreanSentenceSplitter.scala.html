19 package org.openkoreantext.processor.tokenizer

21 case class Sentence(text: String, start: Int, end: Int) {
22   override def toString: String = s"$text($start,$end)"
23 }
24 
25 /**
26  * Sentence Splitter
27  */
28 object KoreanSentenceSplitter {
29   private val re =
30     """(?x)[^.!?…\s]   # First char is non-punct, non-ws
31       [^.!?…]*         # Greedily consume up to punctuation.
32       (?:              # Group for unrolling the loop.
33         [.!?…]         # (special) inner punctuation ok if
34         (?!['\"]?\s|$) # not followed by ws or EOS.
35         [^.!?…]*       # Greedily consume up to punctuation.
36       )*               # Zero or more (special normal*)
37       [.!?…]?          # Optional ending punctuation.
38       ['\"]?           # Optional closing quote.
39       (?=\s|$)""".r
40 
41   def split(s: CharSequence) = {
42     re.findAllMatchIn(s)
43       .map(m => Sentence(m.group(0), m.start(0), m.end(0)))
44       .toSeq
45   }
46 }
