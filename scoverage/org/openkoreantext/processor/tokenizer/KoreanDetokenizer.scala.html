24 /**
25   * Detokenizes a list of tokenized words into a readable sentence.
26   */
27 object KoreanDetokenizer {
28   val SuffixPos = Set(KoreanPos.Josa, KoreanPos.Eomi, KoreanPos.PreEomi, KoreanPos.Suffix,
29     KoreanPos.Punctuation)
30   val PrefixPos = Set(KoreanPos.Modifier, KoreanPos.VerbPrefix)
31 
32   def detokenize(input: Iterable[String]) = {
33     // Space guide prevents tokenizing a word that was not tokenized in the input.
34     val spaceGuide: Set[Int] = getSpaceGuide(input)
35 
36     // Tokenize a merged text with the space guide.
37     val tokenized = KoreanTokenizer.tokenize(input.mkString(""), TokenizerProfile(spaceGuide = spaceGuide))
38 
39     // Attach suffixes and prefixes.
40     // Attach Noun + Verb
41     collapseTokens(tokenized).mkString(" ")
42   }
43 
44   private def collapseTokens(tokenized: Seq[KoreanToken]): List[String] = {
45     val (output, isPrefix, prev) = tokenized
46       .foldLeft[(List[String], Boolean, Option[KoreanToken])]((List[String](), false, None)) {
47       case ((output: List[String], isPrefix: Boolean, prev: Option[KoreanToken]), token: KoreanToken) =>
48         if (output.nonEmpty && (isPrefix || SuffixPos.contains(token.pos))) {
49           val attached = output.lastOption.getOrElse("") + token.text
50           (output.init :+ attached, false, Some(token))
51         } else if (prev.isDefined && prev.get.pos == KoreanPos.Noun && token.pos == KoreanPos.Verb) {
52           val attached = output.lastOption.getOrElse("") + token.text
53           (output.init :+ attached, false, Some(token))
54         } else if (PrefixPos.contains(token.pos)) {
55           (output :+ token.text, true, Some(token))
56         } else {
57           (output :+ token.text, false, Some(token))
58         }
59     }
60     output
61   }
62 
63   private def getSpaceGuide(input: Iterable[String]): Set[Int] = {
64     val (spaceGuide, index) = input.foldLeft((Set[Int](), 0)) {
65       case ((output: Set[Int], i: Int), word: String) =>
66         (output + (i + word.length), i + word.length)
67     }
68     spaceGuide
69   }
70 }
