30 /**
31   * OpenKoreanTextProcessor provides error and slang tolerant Korean tokenization.
32   */
33 object OpenKoreanTextProcessor {
34   /**
35     * Normalize Korean text. Uses KoreanNormalizer.normalize().
36     *
37     * @param text Input text
38     * @return Normalized Korean text
39     */
40   def normalize(text: CharSequence): CharSequence = KoreanNormalizer.normalize(text)
41 
42 
43   /**
44     * Tokenize text into a sequence of KoreanTokens, which includes part-of-speech information and
45     * whether a token is an out-of-vocabulary term.
46     *
47     * @param text input text
48     * @return A sequence of KoreanTokens.
49     */
50   def tokenize(text: CharSequence): Seq[KoreanToken] = KoreanTokenizer.tokenize(text)
51 
52   /**
53     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
54     * which includes part-of-speech information and whether a token is an out-of-vocabulary term.
55     *
56     * @param text input text
57     * @param profile TokenizerProfile
58     * @return A sequence of KoreanTokens.
59     */
60   def tokenize(text: CharSequence,
61                profile: TokenizerProfile
62               ): Seq[KoreanToken] = {
63     KoreanTokenizer.tokenize(text, profile)
64   }
65 
66   /**
67     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
68     * which includes part-of-speech information and whether a token is an out-of-vocabulary term,
69     * and return top `n` candidates.
70     *
71     * @param text input text
72     * @param n number of top candidates
73     * @return A sequence of sequences of KoreanTokens.
74     */
75   def tokenizeTopN(text: CharSequence,
76                    n: Int
77                   ): Seq[Seq[Seq[KoreanToken]]] = {
78     KoreanTokenizer.tokenizeTopN(text, n)
79   }
80 
81   /**
82     * Tokenize text (with a custom profile) into a sequence of KoreanTokens,
83     * which includes part-of-speech information and whether a token is an out-of-vocabulary term,
84     * and return top `n` candidates.
85     *
86     * @param text input text
87     * @param n number of top candidates
88     * @param profile TokenizerProfile
89     * @return A sequence of sequences of KoreanTokens.
90     */
91   def tokenizeTopN(text: CharSequence,
92                    n: Int,
93                    profile: TokenizerProfile
94                   ): Seq[Seq[Seq[KoreanToken]]] = {
95     KoreanTokenizer.tokenizeTopN(text, n, profile)
96   }
97 
98   /**
99     * Add user-defined word list to the noun dictionary. Spaced words are not allowed.
100     *
101     * @param words Sequence of words to add.
102     */
103   def addNounsToDictionary(words: Seq[String]): Boolean = {
104     KoreanDictionaryProvider.addWordsToDictionary(KoreanPos.Noun, words)
105   }
106 
107   /**
108     * Add user-defined word List to the dictionary for the specified KoreanPos.
109     *
110     * @param pos KoreanPos of words to add.
111     * @param words Sequence of words to add.
112     */
113   def addWordsToDictionary(pos: KoreanPos, words: Seq[String]): Boolean = {
114     KoreanDictionaryProvider.addWordsToDictionary(pos, words)
115   }
116 
117   /**
118     * Remove user-defined word List from the dictionary for the specified KoreanPos.
119     *
120     * @param pos KoreanPos of words to add.
121     * @param words Sequence of words to add.
122     */
123   def removeWordsFromDictionary(pos: KoreanPos, words: Seq[String]): Boolean = {
124     KoreanDictionaryProvider.removeWordsToDictionary(pos, words)
125   }
126 
127   /**
128     * Tokenize text into a sequence of token strings. This excludes spaces.
129     *
130     * @param tokens Korean tokens
131     * @return A sequence of token strings.
132     */
133   def tokensToStrings(tokens: Seq[KoreanToken]): Seq[String] = {
134     tokens.filterNot(t => t.pos == KoreanPos.Space).map(_.text.toString)
135   }
136 
137   /**
138     * Split input text into sentences.
139     *
140     * @param text input text
141     * @return A sequence of sentences.
142     */
143   def splitSentences(text: CharSequence): Seq[Sentence] = {
144     KoreanSentenceSplitter.split(text)
145   }
146 
147   /**
148     * Extract noun-phrases from Korean text
149     *
150     * @param tokens         Korean tokens
151     * @param filterSpam     true if spam/slang terms to be filtered out (default: false)
152     * @param enableHashtags true if #hashtags to be included (default: true)
153     * @return A sequence of extracted phrases
154     */
155   def extractPhrases(tokens: Seq[KoreanToken],
156                      filterSpam: Boolean = false,
157                      enableHashtags: Boolean = true): Seq[KoreanPhrase] = {
158     KoreanPhraseExtractor.extractPhrases(tokens, filterSpam, enableHashtags)
159   }
160 
161   /**
162     * Detokenize the input list of words.
163     *
164     * @param tokens List of words.
165     * @return Detokenized string.
166     */
167   def detokenize(tokens: Iterable[String]): String = {
168     KoreanDetokenizer.detokenize(tokens)
169   }
170 }
